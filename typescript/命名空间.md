命名空间用于记录代码类型，同时解决命名冲突。

## 使用
```
namespace Test {
  const a = 1;

  export function fn() {
    return a;
  }
}

console.log(a); // Error，全局找不到a变量
console.log(Test.fn());

// 命名空间例子编译结果
var Test;
(function (Test) {
  const a = 1;
  function fn() {
    return a;
  }
  Test.fn = fn;
})(Test || (Test = {}));
```
